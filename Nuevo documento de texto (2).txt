import os

# Lista de todos los archivos que necesitamos crear
archivos = [
    "app/models/usuario.py",
    "app/models/paciente.py",
    "app/models/medico.py",
    "app/models/cita.py",
    "app/models/consulta.py",
    "app/models/comprobante.py",
    "app/schemas/usuario.py",
    "app/schemas/paciente.py",
    "app/schemas/medico.py",
    "app/schemas/cita.py",
    "app/schemas/consulta.py",
    "app/repositories/usuario_repository.py",
    "app/repositories/paciente_repository.py",
    "app/repositories/medico_repository.py",
    "app/repositories/cita_repository.py",
    "app/services/usuario_service.py",
    "app/services/paciente_service.py",
    "app/services/medico_service.py",
    "app/services/cita_service.py",
    "app/controllers/usuario_controller.py",
    "app/controllers/paciente_controller.py",
    "app/controllers/medico_controller.py",
    "app/controllers/cita_controller.py",
]

print("🚀 Empezando a crear archivos...")

for archivo in archivos:
    # Crear las carpetas si no existen
    os.makedirs(os.path.dirname(archivo), exist_ok=True)
    
    # Crear el archivo solo si no existe
    if not os.path.exists(archivo):
        with open(archivo, "w", encoding="utf-8") as f:
            f.write(f"# Archivo generado automáticamente: {archivo}\n")
        print(f"✅ Creado: {archivo}")
    else:
        print(f"⚠️ Ya existe: {archivo}")

print("\n ¡Listo! Todos los archivos han sido creados.")
input("Presiona Enter para salir...")
